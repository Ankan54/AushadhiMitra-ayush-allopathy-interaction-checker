## PHASE 3: New Research Agent Lambda (R4)

**New File**: `C:/Users/ANKAN/WORK/AushadhiMitra-ayush-allopathy-interaction-checker/lambda/research_tools/handler.py`

This Lambda provides a `search_research_articles` function that wraps the web_search Lambda's Tavily call with research-domain restrictions. It loads domains from S3.

```python
"""
Research Tools Action Group Lambda
Triggered by: Bedrock Research Agent
Functions: search_research_articles, search_clinical_evidence

Searches reputed research domains for drug interaction articles.
Domain restrictions loaded from S3 reference/search_domains.json.
"""
import json
import sys
import os
import logging

logger = logging.getLogger()
logger.setLevel(logging.INFO)

sys.path.insert(0, "/opt/python")
sys.path.insert(0, "/var/task")

import boto3
from shared.bedrock_utils import bedrock_response

S3_BUCKET = os.environ.get("S3_BUCKET", "ausadhi-mitra-667736132441")
TAVILY_SECRET_ARN = os.environ.get("TAVILY_SECRET_NAME", "ausadhi-mitra/tavily-api-key")
TAVILY_API_URL = "https://api.tavily.com/search"

s3 = boto3.client("s3")
secrets_client = boto3.client("secretsmanager")

_tavily_key_cache = None
_domain_config_cache = None


def lambda_handler(event, context):
    logger.info(f"Event: {json.dumps(event, default=str)}")
    action_group = event.get("actionGroup", "research_tools")
    function = event.get("function", "")
    params = {p["name"]: p["value"] for p in event.get("parameters", [])}

    try:
        if function == "search_research_articles":
            result = search_research_articles(
                params.get("query", ""),
                params.get("search_depth", "advanced"),
                int(params.get("max_results", "5")),
            )
        elif function == "search_clinical_evidence":
            result = search_clinical_evidence(
                params.get("ayush_name", ""),
                params.get("allopathy_name", ""),
                params.get("interaction_type", ""),
            )
        else:
            result = {"error": f"Unknown function: {function}"}
    except Exception as e:
        logger.exception(f"Error in {function}")
        result = {"error": str(e)}

    return bedrock_response(action_group, function, result)


def _get_tavily_key() -> str:
    global _tavily_key_cache
    if _tavily_key_cache:
        return _tavily_key_cache
    resp = secrets_client.get_secret_value(SecretId=TAVILY_SECRET_ARN)
    secret = resp["SecretString"]
    try:
        data = json.loads(secret)
        _tavily_key_cache = data.get("api_key") or data.get("TAVILY_API_KEY") or secret
    except json.JSONDecodeError:
        _tavily_key_cache = secret.strip()
    return _tavily_key_cache


def _load_research_domains() -> list:
    global _domain_config_cache
    if _domain_config_cache is not None:
        return _domain_config_cache
    try:
        resp = s3.get_object(Bucket=S3_BUCKET, Key="reference/search_domains.json")
        config = json.loads(resp["Body"].read().decode("utf-8"))
        _domain_config_cache = config.get("research_domains", [])
        return _domain_config_cache
    except Exception as e:
        logger.error(f"Failed to load search_domains.json: {e}")
        return []


def _tavily_search(query: str, search_depth: str, max_results: int, domains: list) -> dict:
    """Execute a Tavily search with domain restrictions."""
    from urllib.request import Request, urlopen
    from urllib.error import URLError

    api_key = _get_tavily_key()
    payload_dict = {
        "api_key": api_key,
        "query": query,
        "search_depth": search_depth,
        "max_results": max_results,
        "include_answer": True,
        "include_raw_content": False,
    }
    if domains:
        payload_dict["include_domains"] = domains

    payload = json.dumps(payload_dict).encode("utf-8")
    req = Request(TAVILY_API_URL, data=payload,
                  headers={"Content-Type": "application/json"}, method="POST")
    try:
        with urlopen(req, timeout=30) as resp:
            return json.loads(resp.read().decode("utf-8"))
    except URLError as e:
        logger.error(f"Tavily API error: {e}")
        return {"results": [], "answer": ""}


def search_research_articles(query: str, search_depth: str = "advanced",
                              max_results: int = 5) -> dict:
    """Search reputed research domains for drug interaction articles."""
    if not query:
        return {"success": False, "message": "query is required"}

    domains = _load_research_domains()
    data = _tavily_search(query, search_depth, max_results, domains)

    sources = []
    for r in data.get("results", []):
        sources.append({
            "url": r.get("url", ""),
            "title": r.get("title", ""),
            "snippet": r.get("content", "")[:500],
            "score": r.get("score", 0),
            "category": "research_paper",
        })

    return {
        "success": True,
        "query": query,
        "answer": data.get("answer", ""),
        "sources": sources,
        "total_results": len(sources),
        "domains_used": domains,
    }


def search_clinical_evidence(ayush_name: str, allopathy_name: str,
                              interaction_type: str = "") -> dict:
    """Search for clinical evidence of a specific drug interaction."""
    if not ayush_name or not allopathy_name:
        return {"success": False, "message": "ayush_name and allopathy_name required"}

    queries = [
        f"{ayush_name} {allopathy_name} drug interaction",
        f"{ayush_name} {allopathy_name} clinical study",
    ]
    if interaction_type:
        queries.append(f"{ayush_name} {interaction_type} {allopathy_name}")

    domains = _load_research_domains()
    all_sources = []
    all_answers = []

    for q in queries:
        data = _tavily_search(q, "advanced", 3, domains)
        if data.get("answer"):
            all_answers.append(data["answer"])
        for r in data.get("results", []):
            url = r.get("url", "")
            if not any(s["url"] == url for s in all_sources):
                all_sources.append({
                    "url": url,
                    "title": r.get("title", ""),
                    "snippet": r.get("content", "")[:500],
                    "score": r.get("score", 0),
                    "category": "research_paper",
                })

    return {
        "success": True,
        "ayush_name": ayush_name,
        "allopathy_name": allopathy_name,
        "answer_summary": " ".join(all_answers)[:2000],
        "sources": all_sources,
        "total_results": len(all_sources),
        "queries_used": queries,
    }
```

def validate_ayush_drug(plant_name: str) -> dict:
    """Check if the AYUSH drug exists in DynamoDB. If not, return error with available list."""
    if not plant_name:
        return {"valid": False, "message": "plant_name is required", "available_plants": []}

    mappings = _load_name_mappings()
    query = plant_name.lower().strip()

    for plant in mappings.get("plants", []):
        all_names = (
            [plant.get("scientific_name", "").lower()]
            + [n.lower() for n in plant.get("common_names", [])]
            + [n.lower() for n in plant.get("hindi_names", [])]
        )
        if query in all_names or any(query in n for n in all_names):
            pk = plant["scientific_name"].replace(" ", "_").lower()
            # Verify DynamoDB entry exists
            dynamodb = boto3.resource("dynamodb")
            table = dynamodb.Table(os.environ.get("DYNAMODB_TABLE", "ausadhi-imppat"))
            try:
                resp = table.get_item(Key={"plant_name": pk, "record_key": "METADATA"})
                if resp.get("Item"):
                    return {
                        "valid": True,
                        "scientific_name": plant["scientific_name"],
                        "dynamodb_key": pk,
                        "common_names": plant.get("common_names", []),
                        "key_phytochemicals": plant.get("key_phytochemicals", []),
                        "known_cyp_effects": plant.get("known_cyp_effects", {}),
                    }
            except Exception as e:
                logger.error(f"DynamoDB validation failed: {e}")

    available = [p["scientific_name"] for p in mappings.get("plants", [])]
    return {
        "valid": False,
        "query": plant_name,
        "message": f"AYUSH drug '{plant_name}' not found in our database.",
        "available_plants": available,

def generate_structured_plan(ayush_name: str, allopathy_name: str,
                              iteration: int = 0, feedback: str = "") -> dict:
    """Generate a structured plan for AYUSH, Allopathy, and Research agents."""
    mappings = _load_name_mappings()
    query_lower = ayush_name.lower().strip()

    plant_data = {}
    for plant in mappings.get("plants", []):
        all_names = (
            [plant.get("scientific_name", "").lower()]
            + [n.lower() for n in plant.get("common_names", [])]
            + [n.lower() for n in plant.get("hindi_names", [])]
        )
        if query_lower in all_names or any(query_lower in n for n in all_names):
            plant_data = plant
            break

    scientific_name = plant_data.get("scientific_name", ayush_name)
    cyp_effects = plant_data.get("known_cyp_effects", {})
    focus_enzymes = list(cyp_effects.keys()) or ["CYP3A4", "CYP2C9", "CYP2D6"]

    if iteration > 0 and feedback:
        # Re-planning based on Reasoning Agent feedback
        try:
            fb = json.loads(feedback) if isinstance(feedback, str) else feedback
        except (json.JSONDecodeError, TypeError):
            fb = {}

        missing = fb.get("missing_data", [])
        plan = {
            "status": "retry",
            "iteration": iteration,
            "ayush_drug": {"scientific_name": scientific_name},
            "allopathy_drug": {"drug_name": allopathy_name},
            "ayush_agent_plan": {"tasks": [], "focus_cyp_enzymes": focus_enzymes},
            "allopathy_agent_plan": {"tasks": [], "focus_cyp_enzymes": focus_enzymes},
            "research_agent_plan": {"tasks": [], "search_queries": []},
        }

        for item in missing:
            agent = item.get("agent", "").lower()
            suggestion = item.get("query_suggestion", "")
            if "ayush" in agent:
                plan["ayush_agent_plan"]["tasks"].append(suggestion)
            elif "allopathy" in agent:
                plan["allopathy_agent_plan"]["tasks"].append(suggestion)
            elif "research" in agent:
                plan["research_agent_plan"]["tasks"].append(suggestion)
                plan["research_agent_plan"]["search_queries"].append(suggestion)

        return {"success": True, "plan": plan}

    # Initial plan
    plan = {
        "status": "proceed",
        "ayush_drug": {
            "scientific_name": scientific_name,
            "dynamodb_key": scientific_name.replace(" ", "_").lower(),
            "common_names": plant_data.get("common_names", []),
        },
        "allopathy_drug": {
            "drug_name": allopathy_name,
            "generic_name": allopathy_name.capitalize(),
        },
        "ayush_agent_plan": {
            "tasks": [
                f"Look up IMPPAT data for {scientific_name}",
                "Find CYP enzyme interactions for all phytochemicals",
                f"Search for known pharmacological effects relevant to drug interactions",
            ],
            "focus_cyp_enzymes": focus_enzymes,
            "focus_properties": plant_data.get("effect_categories", []),
        },
        "allopathy_agent_plan": {
            "tasks": [
                f"Look up {allopathy_name} metabolism data from DrugBank",
                "Check NTI status",
                "Find CYP substrate/inhibitor/inducer profile",
            ],
            "focus_cyp_enzymes": focus_enzymes,
        },
        "research_agent_plan": {
            "tasks": [
                f"Search PubMed for '{scientific_name} {allopathy_name} interaction'",
                f"Search for clinical studies on {scientific_name} CYP effects",
                f"Search for case reports of {scientific_name}-{allopathy_name} interactions",
            ],
            "search_queries": [
                f"{scientific_name} {allopathy_name} drug interaction",
                f"{scientific_name} CYP enzyme inhibition clinical study",
                f"{scientific_name} {allopathy_name} case report",
            ],
        },
    }

    return {"success": True, "plan": plan}

```python
def run_interaction_pipeline(ayush_name: str, allopathy_name: str,
                             session_id: str = None) -> Generator[Tuple[str, Any], None, None]:
    if not session_id:
        session_id = str(uuid.uuid4())

    try:
        ayush_name, allopathy_name = validate_drug_names(ayush_name, allopathy_name)
    except ValueError as e:
        yield ("error", {"message": f"Input validation failed: {e}"})
        return

    query = f"Check interaction between {ayush_name} and {allopathy_name}"
    total_steps = 5  # Was 4, now 5 with Research Agent

    # Step 1: Planner — Validate AYUSH + Generate Plan
    yield ("trace", {"type": "pipeline_step", "agent": "Pipeline",
                     "message": f"Step 1/{total_steps}: AYUSH Validation & Planning"})

    planner_response = ""
    for event_type, data in _invoke_agent("planner", query, session_id):
        yield (event_type, data)
        if event_type == "agent_complete":
            planner_response = data["response"]
        if event_type == "error":
            return

    if not planner_response:
        yield ("error", {"message": "Planner Agent returned empty response"})
        return

    # Check for AYUSH validation failure or cached result
    try:
        plan = json.loads(planner_response)
        if plan.get("cached", False):
            yield ("response", planner_response)
            yield ("done", {"full_response": planner_response, "session_id": session_id,
                            "cached": True, "execution_mode": "pipeline_fallback"})
            return
        # Check if AYUSH drug was not found
        plan_inner = plan.get("plan", plan)
        if plan_inner.get("valid") == False or plan_inner.get("status") == "error":
            error_msg = plan_inner.get("message", "AYUSH drug not found in database")
            available = plan_inner.get("available_plants", [])
            yield ("error", {
                "message": error_msg,
                "available_plants": available,
                "ayush_not_found": True,
            })
            return
    except json.JSONDecodeError:
        pass

    # Step 2: AYUSH Data Collection
    yield ("trace", {"type": "pipeline_step", "agent": "Pipeline",
                     "message": f"Step 2/{total_steps}: AYUSH Phytochemical & CYP Data"})
    ayush_prompt = (f"Based on this plan, gather phytochemical and CYP data:\n\n"
                    f"Plan: {planner_response[:2000]}\n\nFocus on AYUSH substance CYP profile.")
    ayush_response = ""
    for event_type, data in _invoke_agent("ayush", ayush_prompt, session_id):
        yield (event_type, data)
        if event_type == "agent_complete":
            ayush_response = data["response"]

    # Step 3: Allopathy Data Collection
    yield ("trace", {"type": "pipeline_step", "agent": "Pipeline",
                     "message": f"Step 3/{total_steps}: Allopathy Drug Metabolism & NTI Data"})
    allopathy_prompt = (f"Based on this plan, gather drug metabolism and CYP data:\n\n"
                        f"Plan: {planner_response[:2000]}\n\nFocus on CYP pathways and NTI status.")
    allopathy_response = ""
    for event_type, data in _invoke_agent("allopathy", allopathy_prompt, session_id):
        yield (event_type, data)
        if event_type == "agent_complete":
            allopathy_response = data["response"]

    # Step 4: Research Evidence Collection (NEW)
    yield ("trace", {"type": "pipeline_step", "agent": "Pipeline",
                     "message": f"Step 4/{total_steps}: Research Evidence Collection"})
    research_prompt = (f"Based on this plan, search for published research:\n\n"
                       f"Plan: {planner_response[:2000]}\n\nSearch for clinical evidence.")
    research_response = ""
    for event_type, data in _invoke_agent("research", research_prompt, session_id):
        yield (event_type, data)
        if event_type == "agent_complete":
            research_response = data["response"]

    # Step 5: Reasoning with Validation Loop
    # Accumulate data across iterations
    accumulated_ayush = ayush_response
    accumulated_allopathy = allopathy_response
    accumulated_research = research_response

    reasoning_response = ""
    validation = {}
    validation_iteration = 0

    while validation_iteration <= MAX_VALIDATION_RETRIES:
        is_retry = validation_iteration > 0
        step_label = (f"Validation Retry {validation_iteration}/{MAX_VALIDATION_RETRIES}"
                      if is_retry else f"Step 5/{total_steps}: Interaction Reasoning & Validation")
        yield ("trace", {"type": "pipeline_step", "agent": "Pipeline", "message": step_label})

        if is_retry:
            reasoning_prompt = (
                f"RETRY ANALYSIS — Previous analysis incomplete.\n\n"
                f"PREVIOUS ANALYSIS:\n{reasoning_response[:2000]}\n\n"
                f"ACCUMULATED AYUSH DATA:\n{accumulated_ayush[:3000]}\n\n"
                f"ACCUMULATED ALLOPATHY DATA:\n{accumulated_allopathy[:3000]}\n\n"
                f"ACCUMULATED RESEARCH DATA:\n{accumulated_research[:3000]}\n\n"
                f"Complete the analysis with ALL accumulated data."
            )
        else:
            reasoning_prompt = (
                f"Analyze the drug interaction.\n\nPLAN:\n{planner_response[:1000]}\n\n"
                f"AYUSH DATA:\n{accumulated_ayush[:3000]}\n\n"
                f"ALLOPATHY DATA:\n{accumulated_allopathy[:3000]}\n\n"
                f"RESEARCH DATA:\n{accumulated_research[:3000]}\n\n"
                f"Run all analysis phases."
            )

        reasoning_response = ""
        for event_type, data in _invoke_agent(
            "reasoning", reasoning_prompt,
            f"{session_id}-reasoning-v{validation_iteration}",
        ):
            if event_type == "agent_complete":
                reasoning_response = data["response"]
            yield (event_type, data)
            if event_type == "error":
                return

        if not reasoning_response:
            yield ("error", {"message": "Reasoning Agent returned empty response"})
            return

        validation = _extract_validation(reasoning_response)
        status = validation.get("validation_status", "PASSED")
        score = validation.get("completeness_score", 0)

        yield ("validation", {
            "status": status, "completeness_score": score,
            "iteration": validation_iteration,
            "issues": validation.get("issues", []),
            "missing_data": validation.get("missing_data", []),
        })

        if status == "PASSED":
            break

        if validation_iteration >= MAX_VALIDATION_RETRIES:
            yield ("trace", {"type": "validation_max_retries", "agent": "Validation",
                             "message": f"Max retries reached (score: {score}/100)."})
            break

        # Re-invoke agents based on feedback (APPEND to accumulated data)
        retry_plan = _determine_retry_agents(validation)

        if retry_plan["ayush"]:
            for et, d in _invoke_agent(
                "ayush",
                f"Gather ADDITIONAL data for {ayush_name}. "
                f"Focus: {', '.join(validation.get('issues', [])[:3])}",
                f"{session_id}-ayush-retry-{validation_iteration}",
            ):
                yield (et, d)
                if et == "agent_complete":
                    accumulated_ayush += "\n\n--- ADDITIONAL DATA ---\n" + d["response"]

        if retry_plan["allopathy"]:
            for et, d in _invoke_agent(
                "allopathy",
                f"Gather ADDITIONAL data for {allopathy_name}. "
                f"Focus: {', '.join(validation.get('issues', [])[:3])}",
                f"{session_id}-allopathy-retry-{validation_iteration}",
            ):
                yield (et, d)
                if et == "agent_complete":
                    accumulated_allopathy += "\n\n--- ADDITIONAL DATA ---\n" + d["response"]

        # Always re-invoke research on retry for more evidence
        if retry_plan.get("web_search") or retry_plan.get("research", False):
            queries = [item.get("query_suggestion", "") for item in validation.get("missing_data", [])
                       if "research" in item.get("agent", "").lower() or "web" in item.get("source_agent", "").lower()]
            research_retry_prompt = (
                f"Search for ADDITIONAL research evidence. Focus on:\n"
                + "\n".join(f"- {q}" for q in queries[:5])
            )
            for et, d in _invoke_agent(
                "research", research_retry_prompt,
                f"{session_id}-research-retry-{validation_iteration}",
            ):
                yield (et, d)
                if et == "agent_complete":
                    accumulated_research += "\n\n--- ADDITIONAL DATA ---\n" + d["response"]

        validation_iteration += 1

    # Final output
    if reasoning_response:
        yield ("response", reasoning_response)
        yield ("done", {
            "full_response": reasoning_response,
            "session_id": session_id,
            "cached": False,

def _determine_retry_agents(validation: dict) -> dict:
    retry_plan = {"ayush": False, "allopathy": False, "research": False, "web_search": False, "queries": []}

    for item in validation.get("missing_data", []):
        agent = item.get("agent", item.get("source_agent", "")).lower()
        query = item.get("query_suggestion", "")
        if "ayush" in agent:
            retry_plan["ayush"] = True
        elif "allopathy" in agent:
            retry_plan["allopathy"] = True
        elif "research" in agent:
            retry_plan["research"] = True
        elif "web" in agent:
            retry_plan["web_search"] = True
        if query:
            retry_plan["queries"].append({"agent": agent, "query": query})

    issues_text = " ".join(validation.get("issues", [])).lower()
    if not any([retry_plan["ayush"], retry_plan["allopathy"], retry_plan["research"]]):
        if any(kw in issues_text for kw in ["cyp", "phytochemical", "imppat", "ayush"]):
            retry_plan["ayush"] = True
        if any(kw in issues_text for kw in ["nti", "metabolism", "drug data", "allopathy"]):
            retry_plan["allopathy"] = True
        if any(kw in issues_text for kw in ["source", "pubmed", "evidence", "clinical", "research"]):
            retry_plan["research"] = True

    return retry_plan
```
